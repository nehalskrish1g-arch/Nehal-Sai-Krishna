Prim(Graph, start):
∀u: key[u]=∞, parent[u]=null
key[start] = 0
MST = {}
while MST not full:
u = minKey(not in MST)
MST.add(u)
for v in neighbors(u):
if v∉MST ∧ w(u,v)<key[v]:
key[v] = w(u,v)
parent[v] = u
